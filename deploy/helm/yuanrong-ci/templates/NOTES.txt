=== Secrets (must exist before build pods run) ===

Check:
  kubectl get secret obs-credentials -n {{ .Release.Namespace }}
  kubectl get secret swr-credentials -n {{ .Release.Namespace }}

Create if missing:
  kubectl create secret generic obs-credentials \
    --from-literal=AK=<AK> \
    --from-literal=SK=<SK> \
    --from-literal=access-key-id=<AK> \
    --from-literal=secret-access-key=<SK> \
    -n {{ .Release.Namespace }}

  kubectl create secret generic swr-credentials \
    --from-literal=username=<SWR_USERNAME> \
    --from-literal=password=<SWR_PASSWORD> \
    -n {{ .Release.Namespace }}

=== agent-stack-k8s (deployed separately via upstream chart) ===

  helm upgrade --install agent-stack-k8s oci://ghcr.io/buildkite/helm/agent-stack-k8s \
    --version 0.40.0 \
    -f deploy/helm/agent-stack-k8s-values.yaml \
    --set agentToken="${BUILDKITE_AGENT_TOKEN}" \
    -n {{ .Release.Namespace }}

  helm upgrade --install agent-stack-k8s-arm64 oci://ghcr.io/buildkite/helm/agent-stack-k8s \
    --version 0.40.0 \
    -f deploy/helm/agent-stack-k8s-arm64-values.yaml \
    --set agentToken="${BUILDKITE_AGENT_TOKEN}" \
    -n {{ .Release.Namespace }}
